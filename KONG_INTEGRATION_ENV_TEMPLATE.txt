# Kong Integration - Environment Configuration
# 
# Copy this to .env or set these environment variables
# This file provides all Kong-related configuration options

# ===== Kong Configuration =====

# Kong Admin URL - used for Kong configuration
KONG_ADMIN_URL=http://localhost:8001

# Kong Proxy URL - used by adapters to route requests
# When set, all Duffel API calls will go through Kong
# Leave empty to use direct API (failover behavior)
KONG_PROXY_URL=http://kong:8000

# Enable Kong proxy (optional boolean flag)
# Set to 'true' to enforce Kong routing
USE_KONG_PROXY=true

# ===== Duffel API Configuration =====

# Fallback API base URL (used if KONG_PROXY_URL is not set)
DUFFEL_API_BASE_URL=https://api.duffel.com/air

# Duffel API Keys
# Generate these from Duffel Dashboard: https://app.duffel.com/api
DUFFEL_TEST_API_KEY=your_duffel_test_api_key_here
DUFFEL_PROD_API_KEY=your_duffel_prod_api_key_here

# ===== Kong Service Configuration =====

# Kong Service Name (default: duffel-air-service)
KONG_SERVICE_NAME=duffel-air-service

# Kong Service Host (where requests are proxied to)
KONG_SERVICE_HOST=api.duffel.com

# Kong Service Port
KONG_SERVICE_PORT=443

# Kong Service Path
KONG_SERVICE_PATH=/air

# Kong Service Protocol
KONG_SERVICE_PROTOCOL=https

# ===== Kong Connection Timeouts =====

# Connection timeout (milliseconds)
KONG_CONNECT_TIMEOUT=5000

# Write timeout (milliseconds)
KONG_WRITE_TIMEOUT=10000

# Read timeout (milliseconds)
KONG_READ_TIMEOUT=10000

# ===== Kong Rate Limiting =====

# Rate limit per minute
KONG_RATE_LIMIT_MINUTE=1000

# Rate limit per hour
KONG_RATE_LIMIT_HOUR=10000

# Rate limit policy (local or redis)
KONG_RATE_LIMIT_POLICY=local

# Enable fault tolerance (continue even if rate limit check fails)
KONG_RATE_LIMIT_FAULT_TOLERANT=true

# ===== Kong Database Configuration =====

# Kong Database Type (postgres)
KONG_DATABASE=postgres

# Kong Database Host (in Docker: kong-db service name)
KONG_PG_HOST=kong-db

# Kong Database Port
KONG_PG_PORT=5432

# Kong Database User
KONG_PG_USER=kong

# Kong Database Password
KONG_PG_PASSWORD=kong

# Kong Database Name
KONG_PG_DATABASE=kong

# ===== API Gateway Configuration =====

# API Gateway Port
PORT=3000

# API Gateway Environment
NODE_ENV=development

# API Gateway Database Connection
# For local development with PgBouncer
DATABASE_URL=postgresql://tripalfa:tripalfa_dev@pgbouncer:6432/tripalfa?schema=public

# ===== Routing Strategy =====

# Choose which URL to use for Duffel requests:
# "kong" - Route through Kong (http://kong:8000)
# "direct" - Direct to Duffel API (https://api.duffel.com/air)
# "auto" - Use Kong if available, fallback to direct
DUFFEL_ROUTING_STRATEGY=kong

# ===== Logging Configuration =====

# Kong Log Level
KONG_LOG_LEVEL=notice

# Enable Kong Admin API logs
KONG_ADMIN_ACCESS_LOG=/dev/stdout

# Enable Kong Proxy logs
KONG_PROXY_ACCESS_LOG=/dev/stdout

# Enable Kong error logs
KONG_PROXY_ERROR_LOG=/dev/stderr

# ===== Plugin Configuration =====

# Enable Key Auth plugin
KONG_PLUGIN_KEY_AUTH_ENABLED=true

# Enable Rate Limiting plugin
KONG_PLUGIN_RATE_LIMITING_ENABLED=true

# Enable Request Transformer plugin
KONG_PLUGIN_REQUEST_TRANSFORMER_ENABLED=true

# Enable Correlation ID plugin
KONG_PLUGIN_CORRELATION_ID_ENABLED=true

# Correlation ID Header Name
KONG_CORRELATION_ID_HEADER=X-Correlation-ID

# ===== Konga (Kong Admin UI) Configuration =====

# Konga Database Type
KONGA_DB_ADAPTER=postgres

# Konga Database Host
KONGA_DB_HOST=konga-db

# Konga Database Port
KONGA_DB_PORT=5432

# Konga Database User
KONGA_DB_USER=konga

# Konga Database Password
KONGA_DB_PASSWORD=konga

# Konga Database Name
KONGA_DB_DATABASE=konga

# Konga Environment
KONGA_NODE_ENV=production

# ===== Monitoring & Health Checks =====

# Health check endpoint
HEALTH_CHECK_URL=http://localhost:3000/health

# Kong health check endpoint
KONG_HEALTH_CHECK_URL=http://localhost:8001/status

# Enable detailed logging for debugging
DEBUG_MODE=false

# ===== Feature Flags =====

# Enable Ancillary Services through Kong
FEATURE_ANCILLARY_SERVICES_KONG=true

# Enable Seat Maps through Kong
FEATURE_SEAT_MAPS_KONG=true

# Enable Payment through Kong
FEATURE_PAYMENT_KONG=true

# ===== Docker Compose Integration =====

# When using docker-compose.kong.yml, these variables are automatically set:
# - KONG_PROXY_URL=http://kong:8000 (Kong service inside network)
# - Kong containers auto-pass these env vars
# - API Gateway inherits Kong settings

# To override for specific environments:
# docker-compose -f docker-compose.kong.yml up -d --env-file .env.production

# ===== Validation Checklist =====

# Before starting, verify:
# ✓ KONG_PROXY_URL is set correctly
# ✓ DUFFEL_TEST_API_KEY is valid
# ✓ DATABASE_URL points to valid PostgreSQL
# ✓ Kong containers can reach Duffel API on port 443
# ✓ API Gateway container can reach Kong container
# ✓ All required environment variables are set

# ===== Troubleshooting =====

# If requests fail through Kong:
# 1. Check KONG_PROXY_URL is accessible: curl $KONG_PROXY_URL/status
# 2. Verify API key: echo $DUFFEL_TEST_API_KEY
# 3. Check Kong logs: docker logs tripalfa-kong
# 4. Test without Kong: set KONG_PROXY_URL= and restart

# If API Gateway can't connect to Kong:
# 1. Verify Kong is running: docker ps | grep kong
# 2. Check network: docker network inspect tripalfa-network
# 3. Test from API Gateway: docker exec tripalfa-api-gateway-local curl http://kong:8000/status
